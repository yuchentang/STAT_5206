How connected are airports?¶

We could rank the "connectiveness" of an airport and their associated cities by its reach around the globe. Specifically, how many cities are within 1 direct flight or within 1 transfer flight of each city.

We will perform some basic JOIN operations for this to compare the 2019 vs 2020 data.

To answer these questions, you can use the domestic and international destinations from all US airport from this website-%20All%20Carriers&DB_Short_Name=Air%20Carriers). In particular, we have data from both 2019, T_100_MARKET_DOM_INT_2020.csv, and 2020, T_100_MARKET_DOM_INT_2019.csv.

connected <- function(origin, df) {
    thin_df <- df %>% select(ORIGIN, DEST, DEST_COUNTRY)
    flight1 <- thin_df %>% filter(ORIGIN == origin)
    thin_df2 <- thin_df %>% filter(DEST_COUNTRY == "US")
    names(thin_df2) <- paste0(names(thin_df), ".y")
    flight2 <- merge(flight1, thin_df2, by.x = "DEST",
                     by.y= "ORIGIN.y", all.x = TRUE)
    # filter below cannot be done earlier otherwise the
    # join would miss flights that leave the country and
    # fly back in
    flight1 <- filter(flight1, DEST_COUNTRY == "US")
    return(list(f1=unique(flight1$DEST),
                f2=unique(flight2$DEST.y)))
}

dom_connected <- function(origin, us_dom){
    flight1 <- us_dom %>% filter(ORIGIN == origin)
    dom2 <- us_dom
    names(dom2) <- paste0(names(us_dom), ".y")
    flight2 <- merge(flight1, dom2, by.x = "DEST",
                     by.y= "ORIGIN.y", all.x = TRUE)
    return(list(f1=unique(flight1$DEST), f2=unique(flight2$DEST.y)))
}
